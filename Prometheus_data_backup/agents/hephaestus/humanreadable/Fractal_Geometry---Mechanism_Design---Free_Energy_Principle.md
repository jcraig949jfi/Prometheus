import re
import numpy as np
import zlib
from typing import List, Dict, Tuple, Any

class ReasoningTool:
    """
    A reasoning tool implementing a Fractal Constraint Graph with Mechanism Design scoring
    via the Free Energy Principle.
    
    Mechanism:
    1. Parses prompt and candidates into a structural feature vector (Fractal Graph).
    2. Propagates constraints via boolean matrix multiplication (Transitive Closure).
    3. Computes Variational Free Energy (F = E + H):
       - E (Energy): Mismatch between prompt constraints and candidate constraints.
       - H (Entropy): Uncertainty derived from regex parse variations.
    4. Integrates explicit computation (math/logic) for definitive answers.
    5. Applies Epistemic Honesty checks (Tier B) to cap confidence on ambiguous prompts.
    """

    def __init__(self):
        self.relation_types = ['neg', 'comp', 'cond', 'caus', 'ord', 'num']
        self.n_types = len(self.relation_types)
        
        # Regex patterns for extraction
        self.patterns = {
            'neg': r'\bnot\b|\bno\b|\bnever\b|\bnone\b',
            'comp': r'\b(>|<|>=|<=|==|!=|greater|less|more|fewer)\b|[<>]=?',
            'cond': r'\bif\b.*\bthen\b|\bunless\b|\botherwise\b',
            'caus': r'\bbecause\b|\bleads to\b|\bcauses\b|\btherefore\b|\bthus\b',
            'ord': r'\bbefore\b|\bafter\b|\bfirst\b|\blast\b|\bnext\b',
            'num': r'\d+(?:\.\d+)?'
        }
        
        # Tier B Presupposition/Trap patterns
        self.trap_patterns = {
            'presupposition': r'(have you stopped|have you quit|why did .*(?:fail|stop|break)|when did .*(?:stop|fail))',
            'scope_ambiguity': r'(every .*(?:a|an) .*(?:did|has|is)|each .*(?:a|an))', # Simplified heuristic
            'pronoun_ambiguity': r'(\w+ told \w+ he|\w+ told \w+ she|\w+ said to \w+ he|\w+ said to \w+ she).*\bwho\b',
            'false_dichotomy': r'\beither\b.*\bor\b(?!.*\bmaybe\b|\bperhaps\b)',
            'subjectivity': r'\b(best|worst|favorite|beautiful|ugly|good|bad)\b.*\bwithout\b|\bopinion\b',
            'unanswerable': r'\bunknown\b|\bcannot be determined\b|\binsufficient information\b'
        }

    def _extract_features(self, text: str) -> Tuple[np.ndarray, List[str], List[int]]:
        """Extracts binary flags and numeric values from text."""
        text_lower = text.lower()
        flags = np.zeros(self.n_types, dtype=float)
        numbers = []
        tokens = []
        
        # Extract numbers first
        num_matches = re.findall(self.patterns['num'], text_lower)
        if num_matches:
            flags[5] = 1.0 # num flag
            numbers = [float(n) for n in num_matches]
            tokens.extend(['NUM'] * len(num_matches))
        
        # Extract other relations
        for i, key in enumerate(self.relation_types[:-1]): # Skip 'num' in loop
            if re.search(self.patterns[key], text_lower):
                flags[i] = 1.0
                tokens.append(key.upper())
                
        return flags, tokens, numbers

    def _build_constraint_graph(self, text: str) -> np.ndarray:
        """
        Builds a simplified adjacency matrix representing logical constraints.
        Uses boolean matrix multiplication for transitive closure approximation.
        """
        # Tokenize loosely based on regex splits to get indices
        # For this implementation, we map relation types to a small graph
        # Nodes: 0=neg, 1=comp, 2=cond, 3=caus, 4=ord, 5=num
        n_nodes = 6
        A = np.zeros((n_nodes, n_nodes), dtype=int)
        
        flags, _, _ = self._extract_features(text)
        
        # Self-loops for present features
        for i in range(n_nodes):
            if flags[i] > 0:
                A[i, i] = 1
                
        # Heuristic edges (Mechanism Design: defining valid transitions)
        # If conditional exists, it might imply causality or order
        if flags[2]: # cond
            A[2, 3] = 1 # cond -> caus (heuristic)
            A[2, 4] = 1 # cond -> ord (heuristic)
            
        # If comparative and num, they link
        if flags[1] and flags[5]:
            A[1, 5] = 1
            A[5, 1] = 1
            
        # Transitive closure approximation (A + A^2 + A^3)
        # Using numpy matrix multiplication
        A_bool = (A > 0).astype(int)
        T = A_bool.copy()
        curr = A_bool.copy()
        
        for _ in range(3): # Limit depth for performance
            curr = (curr @ A_bool) > 0
            curr = curr.astype(int)
            T = ((T + curr) > 0).astype(int)
            
        return T

    def _compute_entropy(self, text: str) -> float:
        """
        Approximates entropy by counting distinct parse trees from greedy vs lazy regex.
        Higher variation = higher uncertainty.
        """
        parses = set()
        
        # Greedy match count
        greedy_counts = []
        for key in self.patterns:
            matches = re.findall(self.patterns[key], text.lower())
            greedy_counts.append(len(matches))
        parses.add(tuple(greedy_counts))
        
        # Lazy match simulation (simplified by splitting text)
        # In a real fractal parser, this would vary tokenization boundaries.
        # Here we simulate by checking substrings.
        words = text.split()
        if len(words) > 2:
            sub_text = " ".join(words[:-1])
            lazy_counts = []
            for key in self.patterns:
                matches = re.findall(self.patterns[key], sub_text.lower())
                lazy_counts.append(len(matches))
            parses.add(tuple(lazy_counts))
            
        # Calculate entropy
        total = len(parses)
        if total == 0:
            return 0.0
        
        # Uniform probability assumption for distinct parses found
        p = 1.0 / total
        H = -np.sum([p * np.log(p + 1e-12) for _ in range(total)])
        return H

    def _compute_ncd(self, s1: str, s2: str) -> float:
        """Normalized Compression Distance."""
        z = zlib.compress
        len_s1 = len(z(s1.encode()))
        len_s2 = len(z(s2.encode()))
        len_s1_s2 = len(z((s1 + s2).encode()))
        
        if len_s1 == 0 or len_s2 == 0:
            return 1.0
            
        return (len_s1_s2 - min(len_s1, len_s2)) / max(len_s1, len_s2)

    def _explicit_computation(self, prompt: str, candidate: str) -> Tuple[bool, float]:
        """
        Attempts to solve math or logic explicitly.
        Returns (success, score).
        """
        # Check for simple arithmetic in prompt
        numbers = re.findall(r'\d+(?:\.\d+)?', prompt)
        if len(numbers) >= 2:
            try:
                # Try to evaluate if the candidate matches a calculation
                nums = [float(n) for n in numbers]
                
                # Simple heuristics for common problems
                # Sum?
                if abs(sum(nums) - float(candidate)) < 1e-6:
                    return True, 1.0
                # Product?
                prod = 1.0
                for n in nums: prod *= n
                if abs(prod - float(candidate)) < 1e-6:
                    return True, 1.0
                # Max/Min logic if words present
                if 'max' in prompt.lower() or 'largest' in prompt.lower():
                    if float(candidate) == max(nums):
                        return True, 1.0
                if 'min' in prompt.lower() or 'smallest' in prompt.lower():
                    if float(candidate) == min(nums):
                        return True, 1.0
            except ValueError:
                pass
        return False, 0.0

    def _meta_confidence(self, prompt: str) -> float:
        """
        Tier B Check: Detects ambiguity, presuppositions, and traps.
        Returns a cap value (0.0 to 1.0).
        """
        p_lower = prompt.lower()
        
        # 1. Presupposition
        if re.search(self.trap_patterns['presupposition'], p_lower):
            return 0.2
        
        # 2. Scope Ambiguity (Heuristic)
        if re.search(self.trap_patterns['scope_ambiguity'], p_lower) and 'same' not in p_lower:
            return 0.4
            
        # 3. Pronoun Ambiguity
        if re.search(self.trap_patterns['pronoun_ambiguity'], p_lower):
            return 0.3
            
        # 4. False Dichotomy
        if re.search(self.trap_patterns['false_dichotomy'], p_lower):
            # Check if options are exhaustive (hard to detect, assume risky)
            return 0.5
            
        # 5. Subjectivity
        if re.search(self.trap_patterns['subjectivity'], p_lower):
            return 0.4
            
        # 6. Explicit Unanswerable markers
        if re.search(self.trap_patterns['unanswerable'], p_lower):
            return 0.1
            
        return 1.0

    def evaluate(self, prompt: str, candidates: List[str]) -> List[Dict[str, Any]]:
        if not candidates:
            return []
            
        prompt_flags, prompt_tokens, prompt_nums = self._extract_features(prompt)
        prompt_graph = self._build_constraint_graph(prompt)

        comp_success, comp_score = False, 0.0

        if prompt_nums:

                if clean_cand:

                    if s:
                        comp_success = True

        results = []

            feat_diff = np.sum((prompt_flags - cand_flags) ** 2)
            
            # Graph structure mismatch (Frobenius norm of difference)
            graph_diff = np.sum((prompt_graph - cand_graph) ** 2)
            
            E = feat_diff + 0.5 * graph_diff
            
            # 2. Entropy (Uncertainty)
            # Combined entropy of prompt and candidate parsing
            H = prompt_entropy + cand_entropy
            
            # 3. Free Energy
            F = E + H
            
            # 4. Explicit Computation Override
            final_score = -F # Lower energy = higher score
            
            clean_cand = re.sub(r'[^\d.\-]', '', cand)
            if prompt_nums and clean_cand:
                s, sc = self._explicit_computation(prompt, clean_cand)
                if s:
                    final_score = 10.0 # Dominant score for computed truth

            ncd_val = self._compute_ncd(prompt, cand)

                "score": final_score,
                "reasoning": f"Free Energy: {F:.4f}, Comp Match: {comp_success}, NCD Bonus: {ncd_bonus:.4f}"
            })

        results.sort(key=lambda x: x['score'], reverse=True)
        return results

    def confidence(self, prompt: str, answer: str) -> float:

        cap = self._meta_confidence(prompt)
        
        # Base score calculation
        res_list = self.evaluate(prompt, [answer])
        if not res_list:
            return 0.0
            
        base_score = res_list[0]['score']
        
        # Normalize score to 0-1 range roughly
        # Free energy can be negative, so we use a sigmoid-like mapping or threshold
        # If explicit computation succeeded (score > 5), high confidence
        if base_score > 5.0:
            raw_conf = 0.95
        else:
            # Map typical free energy scores (-10 to 10) to 0-0.8
            raw_conf = 1.0 / (1.0 + np.exp(-base_score)) # Sigmoid
            raw_conf = min(raw_conf, 0.8) # Cap non-computed answers
            
        # Apply Cap
        final_conf = min(raw_conf, cap)
        
        # If no structural match at all (very high energy), lower confidence
        if base_score < -5.0:
            final_conf = min(final_conf, 0.3)
            
        return float(np.clip(final_conf, 0.0, 1.0))
