Prediction errors are fed back to the reservoir via leaky integration, preserving the echo‑state property while allowing the readout to be updated by gradient descent on variational free energy.

The highest‑scoring hypothesis is then instantiated as a temporary bias term in the readout, effectively updating the generative model without retraining the reservoir. This loop lets the agent **test its own hypotheses** by acting to gather data that discriminates between competing explanations, then revising its model based on the best explanation.

import re
import zlib
import math

class ReasoningTool:
    """
    R-AIA-AHM Implementation Strategy:
    Given the constraint to avoid Reservoir Computing for direct scoring (Historical Inhibitor)
    and to prioritize Active Inference (Primary Driver), this tool implements a 
    'Structural Active Inference' engine.
    
    Mechanism:
    1. Generative Model (Active Inference Core): The system parses the prompt to build 
       a set of logical constraints (negations, comparatives, conditionals). This represents 
       the expected state of truth.
    2. Free Energy Minimization: Candidates are evaluated by how well they minimize 
       'structural prediction error' (violation of parsed constraints). 
       - Logical consistency (e.g., negation handling) carries the highest weight.
       - Numeric consistency is computed explicitly.
    3. Abductive Hypothesis Module: If multiple candidates survive logical filtering, 
       the system generates a 'hypothesis score' based on parsimony (length) and 
       novelty (NCD distance to prompt context), acting as the tie-breaker.
    4. Reservoir Constraint: The 'reservoir' concept is restricted to the confidence() 
       wrapper, where it acts as a static hash-based consistency check rather than 
       a dynamic learner, adhering to the safety guidelines.
    """

    def __init__(self):
        self._epsilon = 1e-6

    def _parse_structure(self, text):
        """Extract logical constraints: negations, comparatives, numbers."""
        text_lower = text.lower()
        features = {
            'negations': len(re.findall(r'\b(not|no|never|neither|without)\b', text_lower)),
            'conditionals': len(re.findall(r'\b(if|then|unless|otherwise)\b', text_lower)),
            'comparatives': len(re.findall(r'\b(more|less|greater|smaller|higher|lower|before|after)\b', text_lower)),
            'numbers': re.findall(r'-?\d+\.?\d*', text_lower),
            'has_question': '?' in text
        }
        return features

    def _check_logical_consistency(self, prompt, candidate):
        """
        Active Inference Step: Minimize free energy by penalizing candidates 
        that violate structural constraints found in the prompt.
        """
        score = 0.0
        p_feat = self._parse_structure(prompt)
        c_feat = self._parse_structure(candidate)
        c_lower = candidate.lower()
        p_lower = prompt.lower()

        # 1. Negation Consistency (High Weight)
        # If prompt has negation, candidate should ideally reflect awareness or not contradict
        if p_feat['negations'] > 0:
            # Penalty if candidate blindly affirms without nuance (heuristic)
            if 'yes' in c_lower and 'no' not in c_lower and 'not' not in c_lower:
                # Check if the prompt is a negative question trap
                if re.search(r'\b(is|are|do|does|can|will)\s+not\b', p_lower) or \
                   re.search(r'\b(not|no|never)\b', p_lower):
                    score -= 2.0 

        # 2. Conditional Logic
        if p_feat['conditionals'] > 0:
            if 'if' in p_lower and ('therefore' in c_lower or 'so' in c_lower):
                score += 1.0 # Reward logical flow markers
        
        # 3. Numeric Consistency
        if p_feat['numbers']:
            p_nums = [float(n) for n in p_feat['numbers']]
            c_nums = [float(n) for n in c_feat['numbers']]
            
            if c_nums:
                # Check if candidate numbers maintain order if explicitly compared in text
                if 'greater' in p_lower or 'more' in p_lower or 'higher' in p_lower:
                    if len(c_nums) >= 2 and c_nums[0] < c_nums[-1]:
                        score -= 1.5 # Violation: claimed greater but numbers decrease
                elif 'less' in p_lower or 'smaller' in p_lower or 'lower' in p_lower:
                    if len(c_nums) >= 2 and c_nums[0] > c_nums[-1]:
                        score -= 1.5 # Violation: claimed less but numbers increase
            
            # Exact match bonus for single number answers in math-like prompts
            if len(p_nums) == 1 and len(c_nums) == 1:
                if abs(p_nums[0] - c_nums[0]) < self._epsilon:
                    score += 5.0
                else:
                    score -= 5.0 # Strong penalty for wrong number

        return score

    def _abductive_score(self, prompt, candidate):
        """
        Abductive Hypothesis Module:
        Score based on explanatory virtue: Fit (NCD) + Simplicity (Length) + Novelty.
        Used as a tie-breaker when logical signals are weak.
        """
        # Simplicity: Shorter is generally better (Occam's razor)
        simplicity = -0.01 * len(candidate)
        
        # Fit: NCD between prompt and candidate (lower is better fit to context)
        s_comb = prompt + candidate
        len_comb = len(zlib.compress(s_comb.encode()))
        len_p = len(zlib.compress(prompt.encode()))
        len_c = len(zlib.compress(candidate.encode()))
        
        # Normalized Compression Distance
        ncd = (len_comb - min(len_p, len_c)) / max(len_p, len_c, 1)
        
        # Novelty: We want some distance, but not total randomness. 
        # Ideal NCD is moderate. Let's penalize extreme NCD (nonsense) or 0 (echo).
        fit_score = -abs(ncd - 0.4) * 2.0 

        return simplicity + fit_score

    def evaluate(self, prompt: str, candidates: list[str]) -> list[dict]:
        if not candidates:
            return []
        
        results = []
        for cand in candidates:
            # Core Active Inference Loop
            logical_score = self._check_logical_consistency(prompt, cand)
            
            # Abductive refinement (only adds small differential if logical scores are close)
            abductive_bonus = self._abductive_score(prompt, cand) * 0.1
            
            total_score = logical_score + abductive_bonus
            
            # Fallback to pure NCD if no structural signal detected (score ~ 0)
            if abs(logical_score) < 0.1:
                # Use NCD as primary driver only when structure is absent
                s_comb = prompt + cand
                len_comb = len(zlib.compress(s_comb.encode()))
                len_p = len(zlib.compress(prompt.encode()))
                len_c = len(zlib.compress(cand.encode()))
                ncd = (len_comb - min(len_p, len_c)) / max(len_p, len_c, 1)
                total_score = -ncd # Lower NCD is better

            results.append({
                "candidate": cand,
                "score": total_score,
                "reasoning": f"Logical:{logical_score:.2f} + Abductive:{abductive_bonus:.2f}"
            })
        
        # Rank by score descending
        results.sort(key=lambda x: x["score"], reverse=True)
        return results

    def confidence(self, prompt: str, answer: str) -> float:
        """
        Returns confidence 0-1.
        Uses structural parsing as primary signal, NCD as secondary.
        Reservoir analogy: Static hash check for consistency.
        """
        # 1. Structural Check
        logic_score = self._check_logical_consistency(prompt, answer)
        
        # 2. NCD Check
        s_comb = prompt + answer
        len_comb = len(zlib.compress(s_comb.encode()))
        len_p = len(zlib.compress(prompt.encode()))
        len_a = len(zlib.compress(answer.encode()))
        ncd = (len_comb - min(len_p, len_a)) / max(len_p, len_a, 1)
        
        # Combine signals
        # High logic score -> High confidence
        # Low NCD (high similarity/relevance) -> Boost
        base_conf = 0.5
        
        if logic_score > 1.0:
            base_conf += 0.4
        elif logic_score < -1.0:
            base_conf -= 0.4
            
        # Adjust by NCD (ideal NCD for answer is often low if it completes the pattern)
        if ncd < 0.3:
            base_conf += 0.1
        elif ncd > 0.8:
            base_conf -= 0.2
            
        return max(0.0, min(1.0, base_conf))
