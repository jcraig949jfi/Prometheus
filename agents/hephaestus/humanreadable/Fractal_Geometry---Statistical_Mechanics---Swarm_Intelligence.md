import re
import math
import zlib
from typing import List, Dict, Tuple

class ReasoningTool:
    """
    Multi-Scale Renormalized Swarm Optimizer (MRSO) Approximation.
    
    Mechanism:
    1. Fractal Lattice (IFS): The search space is modeled as a recursive decomposition 
       of the prompt-candidate pair. We simulate "tiles" by analyzing the text at 
       different granularities (whole string, halves, quarters).
    2. Statistical Mechanics: Each tile has an "Energy" E based on structural consistency 
       (negation handling, numeric logic, constraint matching). The probability of a 
       candidate being correct follows a Boltzmann distribution P ~ exp(-E/kT).
    3. Swarm Intelligence: Virtual agents traverse these scales. They deposit "pheromones" 
       (score increments) proportional to the Boltzmann weight of the tile's consistency.
    4. Renormalization: Scores from fine scales are coarse-grained (averaged) to form 
       the global score, allowing the system to detect robust hypotheses that hold 
       across resolutions vs fragile ones that fail at specific scales.
    
    This approach prioritizes structural parsing (logic, numbers, negations) as the 
    primary energy function, using NCD only as a high-temperature noise term for tie-breaking.
    """

    def __init__(self):
        self.kT = 0.5  # Temperature parameter for Boltzmann weighting

    def _structural_energy(self, prompt: str, candidate: str) -> float:
        """
        Calculates the 'Energy' (loss) of a candidate given a prompt.
        Lower energy = better fit. Based on structural parsing rules.
        """
        energy = 0.0
        p_lower = prompt.lower()
        c_lower = candidate.lower()
        
        # 1. Negation Handling (Modus Tollens check)
        negations = ["not", "no", "never", "none", "cannot", "impossible"]
        has_negation = any(n in p_lower for n in negations)
        candidate_negates = any(n in c_lower for n in negations)
        
        # If prompt implies negation but candidate affirms (or vice versa), add energy
        # Simple heuristic: if prompt asks "Which is NOT...", candidate should likely contain negation or exclude target
        if "not" in p_lower.split() or "no" in p_lower.split():
            # Crude check: if prompt says "not X", and candidate is "X", high energy
            # We approximate by checking if candidate contradicts the negative constraint
            if has_negation and not candidate_negates:
                energy += 2.0
            elif not has_negation and candidate_negates:
                energy += 1.0

        # 2. Numeric Evaluation
        # Extract numbers from prompt and candidate
        p_nums = re.findall(r"[-+]?\d*\.?\d+", p_lower)
        c_nums = re.findall(r"[-+]?\d*\.?\d+", c_lower)
        
        if p_nums:
            # If numbers exist, check if candidate preserves logical order or magnitude
            try:
                p_max = max(float(x) for x in p_nums)
                if c_nums:
                    c_val = float(c_nums[0])
                    # Heuristic: If prompt implies comparison (e.g., "greater", "less"), 
                    # check if candidate number aligns. 
                    # Since we don't parse full logic, we penalize if candidate number 
                    # is wildly outside the prompt's numeric range (outlier detection)
                    if c_val > p_max * 10 or c_val < p_max * -10:
                        energy += 1.5
                else:
                    # Prompt has numbers, candidate has none -> potential loss of info
                    energy += 0.5
            except ValueError:
                pass

        # 3. Constraint Propagation (Keyword Overlap with Weight)
        # Extract significant words (length > 4) from prompt
        p_words = set(w for w in re.findall(r'\b\w+\b', p_lower) if len(w) > 4)
        c_words = set(w for w in re.findall(r'\b\w+\b', c_lower) if len(w) > 4)
        
        if p_words:
            overlap = len(p_words & c_words)
            coverage = overlap / len(p_words) if len(p_words) > 0 else 0
            # Low coverage increases energy
            energy += (1.0 - coverage) * 2.0

        # 4. NCD Tie-Breaker (Shortest Description Length principle)
        # Used here as a baseline similarity metric, but weighted low (high T)
        try:
            combined = (prompt + candidate).encode('utf-8')
            comp_len = len(zlib.compress(combined))
            p_len = len(zlib.compress(prompt.encode('utf-8')))
            c_len = len(zlib.compress(candidate.encode('utf-8')))
            # Normalized Compression Distance approximation
            ncd = (comp_len - min(p_len, c_len)) / max(p_len, c_len) if max(p_len, c_len) > 0 else 1.0
            energy += ncd * 0.5 # Low weight
        except:
            energy += 1.0

        return energy

    def _fractal_tile_score(self, prompt: str, candidate: str, depth: int) -> float:
        """
        Recursive evaluation simulating agents on a fractal lattice.
        At each depth, we evaluate a 'tile' (substring or full text) and 
        compute a Boltzmann weight.
        """

            return math.exp(-e / self.kT)
        
        # Recursive step: Split prompt into "tiles" (halves) to simulate scale
        mid = len(prompt) // 2
        tile_1 = prompt[:mid]
        tile_2 = prompt[mid:]
        
        # Agents evaluate sub-tiles
        score_1 = self._fractal_tile_score(tile_1, candidate, depth - 1)
        score_2 = self._fractal_tile_score(tile_2, candidate, depth - 1)
        
        # Renormalization: Coarse-grain by averaging (block-spin average)
        # This preserves statistical guarantees while reducing resolution
        return (score_1 + score_2) / 2.0

                reasoning += "Lower probability due to structural mismatches or high energy."
                
            scored_candidates.append({
                "candidate": cand,
                "score": float(raw_score),
                "reasoning": reasoning
            })
        
        # Sort by score descending
        scored_candidates.sort(key=lambda x: x["score"], reverse=True)
        return scored_candidates

    def confidence(self, prompt: str, answer: str) -> float:
        """
        Returns confidence (0-1) based on the normalized Boltzmann weight.

        return conf
